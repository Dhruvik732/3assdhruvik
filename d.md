```c++
vector<int> dotprod(vector<int> a,vector<int> b){
    vector<int> ans;
    for(int i = 0;i<a.size();i++){
        ans.push_back(a[i]*b[i]);
    }
    return ans;
}
int main(){
    vector<int> a = {1,2,3};
    vector<int> b = {2,3,4};
    vector<int> ans = dotprod(a,b);
    for(int i = 0;i<ans.size();i++){
        cout<<ans[i]<<" ";
    }
}
```
